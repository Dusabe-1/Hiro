1. Find Largest Element in an Array

#include <stdio.h>

int main() {
    int arr[5], i, largest;
    printf("Enter 5 numbers: ");
    for(i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    largest = arr[0];
    for(i = 1; i < 5; i++) {
        if(arr[i] > largest)
            largest = arr[i];
    }
    printf("Largest number is %d\n", largest);
    return 0;
}

2. Calculate Average of Array Elements

#include <stdio.h>

int main() {
    int arr[5], i;
    float sum = 0, avg;
    printf("Enter 5 numbers: ");
    for(i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }
    avg = sum / 5;
    printf("Average = %.2f\n", avg);
    return 0;
}

3. Function to Add Two Numbers

#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int main() {
    int x, y;
    printf("Enter two numbers: ");
    scanf("%d %d", &x, &y);
    printf("Sum = %d\n", add(x, y));
    return 0;
}

4. Write to a File

#include <stdio.h>

int main() {
    FILE *fp;
    fp = fopen("output.txt", "w");
    if(fp == NULL) {
        printf("Error opening file!\n");
        return 1;
    }
    fprintf(fp, "Hello, this is a beginner file writing example!");
    fclose(fp);
    printf("Data written to file successfully.\n");
    return 0;
}

5. Read from a File

#include <stdio.h>

int main() {
    FILE *fp;
    char text[100];
    fp = fopen("output.txt", "r");
    if(fp == NULL) {
        printf("Error opening file!\n");
        return 1;
    }
    fgets(text, sizeof(text), fp);
    printf("File content: %s\n", text);
    fclose(fp);
    return 0;
}
