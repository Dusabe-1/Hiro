C File Handling - Reading Files

This page explains how to read data from files in C using fgetc(), fgets(), and fscanf().

1. Reading Character by Character Using fgetc()

#include <stdio.h>

int main() {
    FILE *fp = fopen("input.txt", "r");
    if (fp == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    int ch;
    while ((ch = fgetc(fp)) != EOF) {
        putchar(ch);
    }

    fclose(fp);
    return 0;
}

Explanation: Reads and prints each character until end of file.

2. Reading Line by Line Using fgets()

#include <stdio.h>

int main() {
    FILE *fp = fopen("input.txt", "r");
    if (fp == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    char line[100];
    while (fgets(line, sizeof(line), fp) != NULL) {
        printf("%s", line);
    }

    fclose(fp);
    return 0;
}

Explanation: Reads and prints each line until end of file.

3. Reading Formatted Data Using fscanf()

#include <stdio.h>

int main() {
    FILE *fp = fopen("data.txt", "r");
    if (fp == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    int id;
    char name[30];
    float score;

    while (fscanf(fp, "%d %s %f", &id, name, &score) == 3) {
        printf("ID: %d, Name: %s, Score: %.2f\n", id, name, score);
    }

    fclose(fp);
    return 0;
}

Explanation: Reads formatted data (int, string, float) repeatedly until no more matches.
