1. Find Length of a String

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    printf("Length of string = %lu\n", strlen(str) - 1); // -1 to remove newline
    return 0;
}

2. Reverse a String

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int len, i;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    len = strlen(str) - 1; // Exclude newline
    printf("Reversed string: ");
    for(i = len - 1; i >= 0; i--) {
        printf("%c", str[i]);
    }
    printf("\n");
    return 0;
}

3. Convert String to Uppercase

#include <stdio.h>
#include <ctype.h>

int main() {
    char str[100];
    int i;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    for(i = 0; str[i] != '\0'; i++) {
        str[i] = toupper(str[i]);
    }
    printf("Uppercase: %s", str);
    return 0;
}

4. Count Characters in a String

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int count = 0, i;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    for(i = 0; str[i] != '\0'; i++) {
        if(str[i] != ' ' && str[i] != '\n')
            count++;
    }
    printf("Number of characters (excluding spaces): %d\n", count);
    return 0;
}

5. Sort an Array of Integers

#include <stdio.h>

int main() {
    int arr[5], i, j, temp;
    printf("Enter 5 integers: ");
    for(i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    for(i = 0; i < 4; i++) {
        for(j = i + 1; j < 5; j++) {
            if(arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    printf("Sorted array: ");
    for(i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}
