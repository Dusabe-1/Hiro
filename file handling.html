Simple File Handling (Write & Read)

#include <stdio.h>

int main() {
    FILE *fp;
    char data[100];

    fp = fopen("file.txt", "w");
    if (fp == NULL) {
        printf("Error opening file!\n");
        return 1;
    }
    printf("Enter text to write: ");
    gets(data);
    fputs(data, fp);
    fclose(fp);

    fp = fopen("file.txt", "r");
    if (fp == NULL) {
        printf("Error opening file!\n");
        return 1;
    }
    printf("\nReading from file:\n");
    while (fgets(data, sizeof(data), fp) != NULL)
        printf("%s", data);
    fclose(fp);

    return 0;
}
