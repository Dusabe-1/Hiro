C File Handling - Writing Files

This page explains how to write data to files in C using fprintf() and fputs().

1. Writing Text Using fprintf()

#include <stdio.h>

int main() {
    FILE *fp = fopen("output.txt", "w");
    if (fp == NULL) {
        printf("Unable to open file for writing.\n");
        return 1;
    }
    fprintf(fp, "Hello, this is a line written with fprintf().\n");
    fclose(fp);
    return 0;
}

Explanation: fprintf() works like printf(), but writes formatted output to the file.

2. Writing Text Using fputs()

#include <stdio.h>

int main() {
    FILE *fp = fopen("output.txt", "w");
    if (fp == NULL) {
        printf("Unable to open file for writing.\n");
        return 1;
    }
    fputs("This line is written using fputs().\n", fp);
    fclose(fp);
    return 0;
}

Explanation: fputs() writes a string to the file without formatting.

3. Writing Multiple Lines

#include <stdio.h>

int main() {
    FILE *fp = fopen("output.txt", "w");
    if (fp == NULL) {
        printf("Unable to open file.\n");
        return 1;
    }

    fprintf(fp, "Line 1: Hello!\n");
    fputs("Line 2: Writing files in C.\n", fp);
    fprintf(fp, "Line 3: Goodbye!\n");

    fclose(fp);
    printf("Data written successfully.\n");
    return 0;
}

Output:

Data written successfully.
