C Input Validation

This page explains how to validate user inputs in C to ensure the data is correct and safe.

1. Validate Integer Input Using scanf() Return Value

#include <stdio.h>

int main() {
    int num;
    printf("Enter an integer: ");
    while (scanf("%d", &num) != 1) {
        printf("Invalid input! Please enter an integer: ");
        while (getchar() != '\n'); // clear invalid input
    }
    printf("You entered: %d\n", num);
    return 0;
}

Explanation: scanf() returns the number of inputs successfully read. Loop until an integer is entered.

2. Validate String Input Using fgets() and Check Length

#include <stdio.h>
#include <string.h>

int main() {
    char name[20];
    printf("Enter your name (max 19 chars): ");
    fgets(name, sizeof(name), stdin);
    
    // Remove newline if present
    name[strcspn(name, "\n")] = 0;

    if (strlen(name) == 0) {
        printf("Name cannot be empty.\n");
    } else {
        printf("Hello, %s!\n", name);
    }
    return 0;
}

Explanation: Use fgets() for safe string input and remove newline. Check if input is empty.

3. Validate Float Input

#include <stdio.h>

int main() {
    float value;
    printf("Enter a floating point number: ");
    while (scanf("%f", &value) != 1) {
        printf("Invalid input! Please enter a float: ");
        while (getchar() != '\n'); // clear buffer
    }
    printf("You entered: %.2f\n", value);
    return 0;
}

4. Example: Validate Positive Integer

#include <stdio.h>

int main() {
    int age;
    printf("Enter your age (positive integer): ");
    while (scanf("%d", &age) != 1 || age <= 0) {
        printf("Invalid age! Enter a positive integer: ");
        while (getchar() != '\n');
    }
    printf("Your age is %d\n", age);
    return 0;
}
